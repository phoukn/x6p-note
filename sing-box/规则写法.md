1. 域名后缀匹配（推荐用于直连/代理）
   { "domain_suffix": ["example.com", "google.com"] }
2. 域名关键词匹配
   { "domain_keyword": ["cdn", "api"] }
3. 完整域名精确匹配
   { "domain": ["www.baidu.com", "ipv6.baidu.com"] }
4. IP CIDR 网段匹配（IPv4 / IPv6）
   { "ip_cidr": ["192.168.0.0/16", "10.0.0.0/8", "fc00::/7"] }
5. GeoIP 区域匹配（需下载 geoip.srs）
   { "geoip": ["cn", "private"] }
6. Geosite 域名集匹配（需下载 geosite.srs）
   { "geosite": ["cn", "geolocation-cn"] }
7. 按应用包名（Android UID）分流（仅 TUN 模式有效）
   { "process_name": ["org.telegram.messenger", "com.twitter.android"] }
